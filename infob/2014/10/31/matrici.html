Si consideri il seguente programma, che richiede in ingresso l’inserimento di una matrice 3x3:

#include <stdio.h>
#define DIM 3
int main()
{
  int i,j, M[DIM][DIM];

  int s[3]={0,0,0};

  for (i=0; i<DIM; i++)
      for (j=0; j<DIM; j++)
          scanf("%d",&M[i][j]);

  for (i=0;i<DIM;i++)
  {
      s[0] += M[0][i];
      s[1] += M[1][i];
      s[2] += M[2][i];
  }
  printf("%d %d %d\n", s[0],s[1],s[2]);
  return 0;
}
